Classification Age¶

Add age classifier and return ,store age group in new column age_group

In [89]:
def age_classifier(age):
    if age >= 0 and age <18:
        return "Kids"
    elif age >= 18 and age < 37:
        return "Adults"
    elif age >= 37 and age < 55:
        return "Matures"
    else:
        return "Elders"

df["age_group"] = df["age"].apply(age_classifier)
